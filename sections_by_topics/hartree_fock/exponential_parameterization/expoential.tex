\section{Exponential Parameterization}
    \begin{equation}
        |\Phi _{\text{HF}}\rangle = e ^{-\hat{\kappa}}|0\rangle
    \end{equation}
    in which $\hat{\kappa}$ is an anti-Hermitian operator, i.e.:
    \begin{align}
        \hat{\kappa} = \sum_{p>q}\kappa _{pq}(E _{pq}-&E _{qp}) = \sum_{p>q}\kappa _{pq}E _{pq}^{-} \\
        \bm{\kappa} ^{\dagger} =& -\bm{\kappa}
    \end{align}
