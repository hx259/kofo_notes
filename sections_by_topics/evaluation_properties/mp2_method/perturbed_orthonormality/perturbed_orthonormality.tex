\section{Perturbed Orthonormality Condition}
\subsection{Matrix Parameterization}
We have the general orthonormality condition, subject to perturbation, as:
\begin{equation}
    S _{pq}=\langle p|q\rangle = \delta _{pq}
\end{equation}
\begin{equation}
    \sum_{\mu \nu}C _{\mu p}^{*}S _{\mu \nu}C _{\nu q} = \delta _{pq}
\end{equation}
Parameterization of the MO coefficients:
\begin{equation}
    \mathbf{C}(\lambda) = \mathbf{C}(0)\mathbf{U}(\lambda)
\end{equation}
\begin{equation}
    C _{\mu p}(\lambda) = \sum_{r}C _{\mu r}(0)U _{rp}(\lambda)
\end{equation}
in which $\mathbf{U}(\lambda)$ is the solution to the CPHF equations.
\\

Now the orthonormality condition using this parameterization:
\begin{equation}
    \sum_{\mu \nu}\Big(\sum_{r}U _{rp}^{*}(\lambda)C _{\mu r}^{*}(0)\Big)S _{\mu \nu}(\lambda)\Big(\sum_{s}C _{\nu s}(0)U _{sq}(\lambda)\Big) = \delta _{pq}
\end{equation}
Introducing the transformed overlap matrix:
\begin{equation}
    S ^{00} _{pq}(\lambda) = \sum_{\mu \nu}C _{\mu p}^{*}(0)S _{\mu \nu}(\lambda)C _{\nu q}(0)
\end{equation}
we have:
\begin{equation}
    \sum_{rs}U _{rp}^{*}(\lambda)S ^{00}_{rs}(\lambda)U _{sq}(\lambda) = \delta _{pq}
\end{equation}
differentiating both sides of the equation gives:
\begin{equation}
    \sum_{rs}\frac{\dd{U _{rp}^{*}(\lambda)}}{\dd{\lambda}}S ^{00}_{rs}(\lambda)U _{sq}(\lambda) + \sum_{rs}U _{rp}^{*}(\lambda)\frac{\dd{S ^{00}_{rs}(\lambda)}}{\dd{\lambda}}U _{sq}(\lambda) + \sum_{rs}U _{rp}^{*}(\lambda)S ^{00}_{rs}(\lambda)\frac{\dd{U _{sq}(\lambda)}}{\dd{\lambda}} = 0
\end{equation}
Noting that $\mathbf{S}^{00}(0)=\mathbf{I}$ because the unperturbed spin-orbitals are orthonormal, and it is trivial that $\mathbf{U}(0)=\mathbf{I}$. \\
Therefore evaluating the derivative at $\lambda = 0$, and denoting $A ^{\lambda} =\big(\frac{\dd{A}}{\dd{\lambda}}\big)\big|_{{\lambda=0}}$ results in:
\begin{align}
    \sum_{rs}(U _{rp}^{\lambda})^{*}\delta _{rs}\delta _{sq} + \sum_{rs}\delta _{rp}S_{rs}^{00,\lambda}\delta _{sq} + \sum_{rs}\delta _{rp}\delta _{rs}U _{sq}^{\lambda} = 0
\end{align}
contracting the Kronecker delta tensors we get the perturbed orthonormality condition:
\begin{equation}
    (U _{qp}^{\lambda})^{*} + S_{pq}^{00,\lambda} + U _{pq}^{\lambda} = 0
\end{equation}

\subsection{Exponential Parameterization}
\begin{align}
    \mathbf{C}(\lambda) =& \mathbf{C}(0)\mathbf{U}(\lambda) \\
    \mathbf{U}(\lambda) =& (\mathbf{S}^{00})^{-\frac{1}{2}}(\lambda)\exp[-\bm{\kappa}(\lambda)] \\
    \mathbf{S}^{00}(\lambda) =& \mathbf{C}^{\dagger}(0)\mathbf{S}^{\text{AO}}(\lambda)\mathbf{C}(0)
\end{align}

In this way:
\begin{align}
     \mathbf{S}(\lambda) =& \mathbf{C}^{\dagger}(\lambda)\mathbf{S}^{\text{AO}}(\lambda)\mathbf{C}(\lambda) \nonumber \\
    =& \mathbf{U}^{\dagger}(\lambda)\mathbf{C}^{\dagger}(0)\mathbf{S}^{\text{AO}}(\lambda)\mathbf{C}(0)\mathbf{U}(\lambda) \nonumber \\
    =& \mathbf{U}^{\dagger}(\lambda)(\mathbf{S}^{00})(\lambda)\mathbf{U}(\lambda) \nonumber \\
    =& \exp[-\bm{\kappa}(\lambda)]^{\dagger}(\mathbf{S}^{00})^{-\frac{1}{2} \dagger}(\lambda)\mathbf{S} ^{00}(\lambda)(\mathbf{S}^{00})^{-\frac{1}{2}}(\lambda)\exp[-\bm{\kappa}(\lambda)] \nonumber \\
    =& \exp[\bm{\kappa}(\lambda)]\exp[-\bm{\kappa}(\lambda)] \nonumber \\
    =& \mathbf{I}
\end{align}
the orthonormality is ensured by $(\mathbf{S}^{00})^{-\frac{1}{2}\dagger}\mathbf{S}^{00}(\mathbf{S}^{00})^{-\frac{1}{2}} = \mathbf{I}$, as $\mathbf{S}^{\text{AO}}$ thus $\mathbf{S}^{00}$ matrix is Hermitian. The perturbed orthonormality condition is then trivial:
\begin{equation}
    \mathbf{S}^{\lambda} = \pdv{\mathbf{S}(\lambda)}{\lambda}\Big|_{{\lambda = 0}} = \mathbf{0}
\end{equation}
With this new parameterization, the orthonormality condition and perturbed orthonormality condition come naturally, hence do not need to be included explicitly in the Lagrangian.\\

The derivative of $(\mathbf{S}^{00})^{-\frac{1}{2}}(\lambda)$ (needed for $\mathbf{U}^{\lambda}$ appearing in second derivative) could be solved by taking derivative on this equation:
\begin{equation}
    (\mathbf{S}^{00})^{-\frac{1}{2}\dagger}\mathbf{S}^{00}(\mathbf{S}^{00})^{-\frac{1}{2}} = \mathbf{I}
\end{equation}
i.e. (note that $\mathbf{S}^{00,\dagger} = \mathbf{S}^{00}$ and $\mathbf{S}^{00}(\lambda=0)=\mathbf{I}$ ):
\begin{align}
    \mathbf{0} =& \pdv{(\mathbf{S}^{00}) ^{-\frac{1}{2}}}{\lambda}\mathbf{S}^{00}(\mathbf{S}^{00})^{-\frac{1}{2}}\Big|_{{\lambda=0}} + (\mathbf{S}^{00})^{-\frac{1}{2}}\pdv{\mathbf{S}^{00}}{\lambda} \left(\mathbf{S}^{00}\right)^{-\frac{1}{2}}\Big|_{{\lambda=0}} + (\mathbf{S}^{00})^{-\frac{1}{2}}\mathbf{S}^{00}\pdv{(\mathbf{S}^{00})^{-\frac{1}{2}}}{\lambda}\Big|_{{\lambda=0}} \nonumber \\
    =& 2 (\mathbf{S}^{00})^{-\frac{1}{2},\lambda} + \mathbf{S}^{00,\lambda}
\end{align}
\begin{equation}
    \Leftrightarrow (\mathbf{S}^{00})^{-\frac{1}{2},\lambda} = -\frac{1}{2}\mathbf{S}^{00,\lambda}
\end{equation}
