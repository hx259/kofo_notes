\section{Summary: Derivation of Z-Vector Equation}
\subsection{Notation and Convention}
MP2 Lagrangian:
\begin{align}
    \mathcal{L}_{\text{MP2}} = E _{\text{HF}} + E _{\text{H}} + \frac{1}{2}\sum_{ai}\left(z _{ai}f _{ai} + z _{ai}^{*}f _{ai}^{*}\right)
\end{align}
in which $E _{\text{H}}$ is the Hylleraas functional:
\begin{align}
    E _{\text{H}} =& \sum_{ij}h _{ij}\gamma ^{\text{H}} _{ij} + \sum_{ab}h _{ab}\gamma ^{\text{H}} _{ab} + \sum_{ijab}\langle ab||ij\rangle (\Gamma ^{\text{H}}) _{ij}^{ab}+ \sum_{ijab}\langle ij||ab\rangle (\Gamma ^{\text{H}})^{ij}_{ab} \nonumber \\
    &+\sum_{ijkl}\langle ij||kl\rangle (\Gamma ^{\text{H}}) _{kl}^{ij} + \sum_{ijab}\langle ai||bj\rangle (\Gamma ^{\text{H}}) _{bj}^{ai}
\end{align}
Unrelaxed reduced density matrices (RDMs):
\begin{align}
    \gamma _{pq} =& \gamma ^{\text{HF}} _{pq} + \gamma ^{\text{H}} _{pq} \\
    \Gamma _{rs}^{pq} =& (\Gamma ^{\text{HF}})_{rs}^{pq} + (\Gamma ^{\text{H}})_{rs}^{pq}
\end{align}
\begin{align}
    &\gamma _{ij}^{\text{H}} = \frac{1}{2}\sum_{kab}(T _{jk}^{ab})^{*}T _{ki}^{ab} &&& \gamma ^{\text{HF}}_{ij} = \delta _{ij}&\\
    &\gamma^{\text{H}} _{ab} = -\frac{1}{2}\sum_{ijc}(T _{ij}^{ac})^{*}T _{ij}^{cb} \\
    &(\Gamma ^{\text{H}}) _{ij}^{ab} = \frac{1}{4}(T _{ij}^{ab})^{*} \\
    &(\Gamma ^{\text{H}}) _{ab}^{ij} = \frac{1}{4}T _{ij}^{ab} \\
    &(\Gamma ^{\text{H}}) _{kl}^{ij} = \frac{1}{2}\sum_{mab}(T _{km}^{ab})^{*}T _{mi}^{ab}\delta _{jl}=\gamma _{ik}^{\text{H}}\delta _{jl} &&& (\Gamma ^{\text{HF}})_{kl}^{ij} = \frac{1}{2}\delta _{ik}\delta _{jl}& \\
    &(\Gamma ^{\text{H}}) _{bj}^{ai} = -\frac{1}{2}\sum_{klc}(T _{kl}^{ac})^{*}T _{kl}^{cb} \delta _{ij}=\gamma _{ab}^{\text{H}}\delta _{ij}
\end{align}
The MP2 amplitudes and its symmetry:
\begin{align}
    T _{ij}^{ab} = -T _{ij}^{ba} = -T _{ji}^{ab} = T _{ji}^{ba} = \frac{\langle ab||ij\rangle}{\Delta _{ab}^{ij}}
\end{align}
Constraint on orbital rotation for the Lagrangian (i.e. z-vector equation):
\begin{align}
    \pdv{\mathcal{L}_{\text{MP2}}}{\kappa _{bj}} =& \pdv{E _{\text{H}}}{\kappa _{bj}} + \frac{1}{2}\sum_{ai}\left(z _{ai}\pdv{f _{ai}}{\kappa _{bj}} + z _{ai}^{*}\pdv{f _{ai}^{*}}{\kappa _{bj}}\right) = 0 \\
    \pdv{\mathcal{L}_{\text{MP2}}}{\kappa _{bj}^{*}} =& \pdv{E _{\text{H}}}{\kappa _{bj}^{*}} + \frac{1}{2}\sum_{ai}\left(z _{ai} \pdv{f _{ai}}{\kappa _{bj}^{*}}+z _{ai}^{*}\pdv{f _{ai}^{*}}{\kappa _{bj}^{*}}\right) = 0
\end{align}
The LHS of z-vector equation contains z-vector terms, and the RHS is the derivative of the Hylleraas functional.\\
In this derivation, the following index convention is used:
\begin{itemize}
\item $p,q,r,s,x,y$: general space 
\item $i,j,k,l,m,n$: internal space
\item $a,b,c,d,e,f$: virtual space 
\end{itemize}
*All the derivatives are taken at zero-point of the external perturbation, i.e. $\lambda = 0$, and canonical orbitals are used throughout.

\subsection{LHS}
The LHS of Z-Vector equation:
\begin{align}
    \pdv{f _{ai}}{\kappa _{bj}} =& (\varepsilon _{i}-\varepsilon _{a})\delta _{ij}\delta _{ab} - \langle aj||ib\rangle \\
    \pdv{f _{ai}^{*}}{\kappa _{bj}}=& -\langle ij||ab\rangle \\
    \pdv{f _{ai}}{\kappa _{bj}^{*}} =& -\langle ab||ij\rangle \\
    \pdv{f _{ai}^{*}}{\kappa _{bj}^{*}} =& (\varepsilon _{i}-\varepsilon _{a})\delta _{ij}\delta _{ab} - \langle ib||aj\rangle
\end{align}

Stationarity with $\kappa _{bj}$: 
\begin{align}
    \text{LHS} =& \frac{1}{2}\sum_{ai}\left(z _{ai}\pdv{f _{ai}}{\kappa _{bj}}+ z _{ai}^{*}\pdv{f _{ai}^{*}}{\kappa _{bj}}\right) \nonumber \\
    =&\frac{1}{2}\sum_{ai}z _{ai}\Big((\varepsilon _{i}-\varepsilon _{a})\delta _{ij}\delta _{ab} - \langle aj||ib\rangle\Big) - \frac{1}{2}\sum_{ai}z _{ai}^{*}\langle ij||ab\rangle \nonumber \\
    =& \frac{1}{2}z _{bj}\left(\varepsilon _{j}-\varepsilon _{b}\right)-\frac{1}{2}\sum_{ai}\Big(z _{ai}\langle aj||ib\rangle + z _{ai}^{*}\langle ij||ab\rangle\Big)
\end{align}

\subsection{RHS}
The RHS of Z-Vector equation comes from the Hylleraas response to orbital rotation.
\begin{align}
    \pdv{E _{\text{H}}}{\kappa _{em}} =& \sum_{ij}\pdv{(h _{ij}\gamma ^{\text{H}} _{ij})}{\kappa _{em}} + \sum_{ab}\pdv{(h _{ab}\gamma ^{\text{H}} _{ab})}{\kappa _{em}} + \sum_{ijab}\pdv{(\langle ab||ij\rangle (\Gamma ^{\text{H}}) _{ij}^{ab})}{\kappa _{em}}+ \sum_{ijab}\pdv{(\langle ij||ab\rangle (\Gamma ^{\text{H}})^{ij}_{ab})}{\kappa _{em}} \nonumber \\
    &+\sum_{ijkl}\pdv{(\langle ij||kl\rangle (\Gamma ^{\text{H}}) _{kl}^{ij})}{\kappa _{em}} + \sum_{ijab}\pdv{(\langle ai||bj\rangle (\Gamma ^{\text{H}}) _{bj}^{ai})}{\kappa _{em}}
\end{align}
As the one- and two-body unrelaxed RDMs only depends on MP2 amplitudes, which are treated indepently from the orbital rotation parameter $\bm{\kappa}$, hence only the derivatives of electron integrals against $\bm{\kappa}$ are concerned.
